public class ReMsg implements Serializable{

    private enum code; // [OK_200,BadRequest_400,ForBidden_403,NotFound_404,ToManyRequests_429,ServerError_500]
    private String error; //
    private Object msgMap; //

    public enum getCode(){
        return code;
    }

    public void setCode(enum code){
        this.code = code;
    }

    public String getError(){
        return error;
    }

    public void setError(String error){
        this.error = error;
    }

    public Object getMsgMap(){
        return msgMap;
    }

    public void setMsgMap(Object msgMap){
        this.msgMap = msgMap;
    }

}
